🛠 Proposed Fixes

Fix 1 — Page-level Referrer-Policy (recommended):
Add this meta tag to the page <head> or configure it as an HTTP header on the server. This is the cleanest fix and resolves the issue globally. 

<meta name="referrer" content="strict-origin-when-cross-origin">

Fix 2 — CookiebotOnAccept callback:
If the auto-blocker strips the referrerpolicy attribute during block/unblock, this script re-applies it after consent is given. 

window.addEventListener('CookiebotOnAccept', function() {
  var iframes = document.querySelectorAll(
    'iframe[src*="youtube.com"], iframe[src*="youtube-nocookie.com"]'
  );
  iframes.forEach(function(iframe) {
    if (!iframe.getAttribute('referrerpolicy')) {
      iframe.setAttribute(
        'referrerpolicy',
        'strict-origin-when-cross-origin'
      );
    }
  });
});

Fix 3 — MutationObserver fallback:
For cases where the iframe is dynamically injected after consent (e.g. WP YouTube block), this observer catches new iframes and patches them. 

window.addEventListener('CookiebotOnAccept', function() {
  function patchIframe(iframe) {
    if (iframe.src &&
       (iframe.src.includes('youtube.com') ||
        iframe.src.includes('youtube-nocookie.com')) &&
       !iframe.getAttribute('referrerpolicy')) {
      iframe.setAttribute(
        'referrerpolicy',
        'strict-origin-when-cross-origin'
      );
      // Force reload to apply new policy
      iframe.src = iframe.src;
    }
  }

  // Patch existing iframes
  document.querySelectorAll('iframe').forEach(patchIframe);

  // Watch for new iframes
  var observer = new MutationObserver(function(mutations) {
    mutations.forEach(function(m) {
      m.addedNodes.forEach(function(node) {
        if (node.tagName === 'IFRAME') patchIframe(node);
        if (node.querySelectorAll) {
          node.querySelectorAll('iframe').forEach(patchIframe);
        }
      });
    });
  });
  observer.observe(document.body, {
    childList: true, subtree: true
  });
});
